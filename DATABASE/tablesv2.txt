DROP DATABASE IF EXISTS cavitemed;
CREATE DATABASE cavitemed CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE cavitemed;

-- ----------------------------
-- 1) HEALTH CENTERS
-- ----------------------------
CREATE TABLE health_centers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    center_name VARCHAR(255) NOT NULL,
    center_type ENUM('rhu','barangay_health_center','hospital','clinic','other') NOT NULL,
    address TEXT,
    contact_person VARCHAR(255),
    contact_number VARCHAR(100),
    is_archived TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 2) USERS
-- ----------------------------
CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    full_name VARCHAR(100) NOT NULL,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL DEFAULT 'cavmed2025',
    role ENUM('admin', 'doctor', 'medical_staff', 'encoder') NOT NULL,
    email VARCHAR(100),
    contact_number VARCHAR(20),
    position VARCHAR(100),
    profile_picture VARCHAR(255) NULL,
    status ENUM('active', 'inactive') DEFAULT 'active',
    health_center_id INT NULL,
    last_login DATETIME NULL,
    failed_attempts INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deleted_at DATETIME NULL,
    must_change_password TINYINT(1) DEFAULT 1,

    CONSTRAINT fk_user_health_center
        FOREIGN KEY (health_center_id)
        REFERENCES health_centers(id)
        ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 3) SUPPLIERS
-- ----------------------------
CREATE TABLE suppliers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    supplier_name VARCHAR(255) NOT NULL,
    supplier_type ENUM('private','donation','government') NOT NULL,
    contact_person VARCHAR(255),
    contact_number VARCHAR(100),
    is_archived TINYINT(1) DEFAULT 0,
    email VARCHAR(255),
    address TEXT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 4) MEDICINE
-- ----------------------------
CREATE TABLE medicine (
    id INT AUTO_INCREMENT PRIMARY KEY,
    supplier_id INT NOT NULL,
    barcode VARCHAR(100) UNIQUE NOT NULL,
    medicine_name VARCHAR(255) NOT NULL,
    medicine_type ENUM('generic', 'branded') NOT NULL,
    category VARCHAR(100) NOT NULL,
    funding_source ENUM('donation','city_health','mixed') NOT NULL,
    batch_number VARCHAR(100) NOT NULL,
    expiry_date DATE NOT NULL,
    unit_of_measure ENUM('bottle','tablet','capsule','other') NOT NULL,
    manufacturing_date DATE NOT NULL,
    current_stock INT NOT NULL DEFAULT 0,
    reorder_point INT NOT NULL DEFAULT 0,
    unit_price DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    notes TEXT,
    status ENUM('in_stock','low_stock','out_of_stock','expired') DEFAULT 'in_stock',
    is_archived TINYINT(1) DEFAULT 0,
    created_by INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    CONSTRAINT fk_medicine_supplier
        FOREIGN KEY (supplier_id)
        REFERENCES suppliers(id)
        ON DELETE RESTRICT,

    CONSTRAINT fk_medicine_created_by
        FOREIGN KEY (created_by)
        REFERENCES users(user_id)
        ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 5) INVENTORY TRANSACTIONS
-- ----------------------------
CREATE TABLE inventory_transactions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    medicine_id INT NOT NULL,
    transaction_type ENUM('add','deduct','dispose','distribute','adjustment','expired') NOT NULL,
    quantity INT NOT NULL,
    remarks VARCHAR(255),
    performed_by INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_transaction_medicine
        FOREIGN KEY (medicine_id)
        REFERENCES medicine(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_transaction_performed_by
        FOREIGN KEY (performed_by)
        REFERENCES users(user_id)
        ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 6) DISTRIBUTION
-- ----------------------------
CREATE TABLE distribution (
    id INT AUTO_INCREMENT PRIMARY KEY,
    health_center_id INT NOT NULL,
    medicine_id INT NOT NULL,
    quantity INT NOT NULL,
    status ENUM('pending','distributed','cancelled','returned') DEFAULT 'pending',
    remarks TEXT,
    created_by INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_distribute_health_center
        FOREIGN KEY (health_center_id)
        REFERENCES health_centers(id)
        ON DELETE RESTRICT,

    CONSTRAINT fk_distribute_medicine
        FOREIGN KEY (medicine_id)
        REFERENCES medicine(id)
        ON DELETE RESTRICT,

    CONSTRAINT fk_distribute_created_by
        FOREIGN KEY (created_by)
        REFERENCES users(user_id)
        ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 7) DISPOSAL RECORDS
-- ----------------------------
CREATE TABLE disposal_records (
    id INT AUTO_INCREMENT PRIMARY KEY,
    medicine_id INT NOT NULL,
    batch_number VARCHAR(100) NOT NULL,
    expiry_date DATE NOT NULL,
    quantity INT NOT NULL CHECK (quantity > 0),
    total_value DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    disposal_method ENUM('incinerated','returned','destroyed','donated','other') NOT NULL,
    disposal_date DATE NOT NULL,
    created_by INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    notes TEXT NULL,
    is_archived TINYINT(1) DEFAULT 0,

    CONSTRAINT fk_disposal_medicine
        FOREIGN KEY (medicine_id)
        REFERENCES medicine(id)
        ON DELETE RESTRICT,

    CONSTRAINT fk_disposal_created_by
        FOREIGN KEY (created_by)
        REFERENCES users(user_id)
        ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 8) HEALTH CENTER INVENTORY
-- ----------------------------
CREATE TABLE health_center_inventory (
    id INT AUTO_INCREMENT PRIMARY KEY,
    health_center_id INT NOT NULL,
    medicine_id INT NOT NULL,
    current_stock INT NOT NULL DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY unique_center_medicine (health_center_id, medicine_id),

    CONSTRAINT fk_hci_center
        FOREIGN KEY (health_center_id)
        REFERENCES health_centers(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_hci_medicine
        FOREIGN KEY (medicine_id)
        REFERENCES medicine(id)
        ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 9) STOCK RETURNS
-- ----------------------------
CREATE TABLE stock_returns (
    id INT AUTO_INCREMENT PRIMARY KEY,
    health_center_id INT NOT NULL,
    medicine_id INT NOT NULL,
    quantity INT NOT NULL,
    reason VARCHAR(255) NOT NULL,
    notes TEXT NULL,
    returned_by INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (health_center_id) REFERENCES health_centers(id) ON DELETE CASCADE,
    FOREIGN KEY (medicine_id) REFERENCES medicine(id) ON DELETE CASCADE,
    FOREIGN KEY (returned_by) REFERENCES users(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 10) PATIENTS (ALTER merged here)
-- ----------------------------
CREATE TABLE patients (
  patient_id INT AUTO_INCREMENT PRIMARY KEY,
  mrn VARCHAR(30) NOT NULL UNIQUE, -- Medical Record Number (e.g. 2024-001234)

  first_name VARCHAR(100) NOT NULL,
  last_name VARCHAR(100) NOT NULL,
  middle_name VARCHAR(100) NULL,
  preferred_name VARCHAR(100) NULL,

  marital_status VARCHAR(50) NULL,
  occupation VARCHAR(120) NULL,
  preferred_language VARCHAR(120) NULL,

  date_of_birth DATE NOT NULL,
  gender ENUM('male','female','other') NOT NULL,
  blood_type ENUM('A+','A-','B+','B-','AB+','AB-','O+','O-') NULL,

  phone VARCHAR(30) NOT NULL,
  email VARCHAR(150) NULL,

  address_line VARCHAR(255) NOT NULL,
  city VARCHAR(100) NOT NULL,
  state VARCHAR(100) NOT NULL,
  zip_code VARCHAR(20) NOT NULL,

  status ENUM('active','inactive') DEFAULT 'active',

  health_center_id INT NULL,          -- ✅ NEW: patient scope per center
  created_by INT NOT NULL,

  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  INDEX idx_patients_created_by (created_by),
  INDEX idx_patients_health_center (health_center_id),  -- ✅ NEW

  CONSTRAINT fk_patients_created_by
    FOREIGN KEY (created_by) REFERENCES users(user_id)
    ON DELETE RESTRICT,

  CONSTRAINT fk_patients_health_center       -- ✅ NEW
    FOREIGN KEY (health_center_id) REFERENCES health_centers(id)
    ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ----------------------------
-- 18) PRESCRIPTION ITEMS
-- ----------------------------
CREATE TABLE prescription_items (
  item_id INT AUTO_INCREMENT PRIMARY KEY,

  prescription_id INT NOT NULL,
  medicine_id INT NOT NULL,

  dosage_amount DECIMAL(10,2) NOT NULL,
  dosage_unit ENUM('mg','mcg','g','ml','units') NOT NULL,

  frequency_template ENUM(
    'once-daily','twice-daily','three-times','four-times',
    'every-6h','every-8h','every-12h','as-needed','bedtime'
  ) NOT NULL,

  duration_amount INT NOT NULL,
  duration_unit ENUM('days','weeks','months') NOT NULL,

  route_admin ENUM(
    'oral','sublingual','topical','inhalation',
    'injection-im','injection-iv','injection-sc',
    'rectal','ophthalmic','otic'
  ) NOT NULL,

  item_instructions TEXT NULL,

  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_pres_item_prescription
    FOREIGN KEY (prescription_id) REFERENCES prescriptions(prescription_id)
    ON DELETE CASCADE,

  CONSTRAINT fk_pres_item_medicine
    FOREIGN KEY (medicine_id) REFERENCES medicine(id)
    ON DELETE RESTRICT,

  INDEX idx_pres_item_prescription_id (prescription_id),
  INDEX idx_pres_item_medicine_id (medicine_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
